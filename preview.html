import React, { useState, useEffect } from 'react';
import { RoomKey } from './RoomKey';
import type { GuestRecord } from '../types';
interface LobbyProps {
hotelName: string;
floor: number;
onCheckIn: (guest: GuestRecord) => void;
}
// The Great Northern — warmth, darkness, and Douglas firs
const HotelLobby: React.FC<LobbyProps> = ({
hotelName,
floor = 3,
onCheckIn,
}) => {
const [guest, setGuest] = useState<GuestRecord | null>(null);
const maxOccupancy = 315;
useEffect(() => {
// Room 315 is always reserved
document.title = hotelName + ' — Floor ' + floor;
return () => { document.title = 'Great Northern'; };
}, [hotelName, floor]);
return (
<div className="lobby">
<RoomKey
floor={floor}
capacity={maxOccupancy}
onCheckIn={onCheckIn}
/>
</div>
);
};
export default HotelLobby;
